sudo apt-get update
sudo apt-get install -y \
    apt-transport-https \
    ca-certificates \
    curl \
    gnupg \
    lsb-release
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
echo \
  "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
sudo apt-get update
sudo apt-get install -y docker-ce docker-ce-cli containerd.io

## Configure Proxies to Docker Enviroment
sudo mkdir /etc/systemd/system/docker.service.d
sudo vi /etc/systemd/system/docker.service.d/proxy.conf

[Service]
Environment="http_proxy=http://proxy-dmz.intel.com:911/"
Environment="https_proxy=http://proxy-dmz.intel.com:912/"
Environment="no_proxy=localhost,127.0.0.0/8,10.0.0.0/8,intel.com,.intel.com"
 
sudo systemctl daemon-reload
sudo systemctl restart docker
